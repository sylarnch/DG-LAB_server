log_level : 日志级别

void parseConfig(const std::string& filename) {
    std::ifstream file(filename);
    std::string line;
    while (std::getline(file, line)) {
        auto pos = line.find(':');
        std::string key = line.substr(0, pos);
        std::string value = line.substr(pos + 1);
        std::cout << "Key: " << key << ", Value: " << value << std::endl;
    }
}
